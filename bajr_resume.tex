\newcommand{\checkdepth}[1]{
  \ifnum\value{workdepth}>#1
    \addtocounter{workdepth}{-1}
    \end{itemize}
    \checkdepth{#1} % Keep going until desired depth
  \fi
  \ifnum\value{workdepth}<#1
    \ifnum\value{workdepth}<2
      \begin{itemize}[nosep,itemindent=-1cm,label=]
    \fi
    \ifnum\value{workdepth}=2
      \begin{itemize}[nosep,itemindent=-1cm,label=$\bullet$]
    \fi
    \stepcounter{workdepth}
    \checkdepth{#1} % Keep going until desired depth
  \fi
}
