\newcommand{\checkdepth}[1]{% Argument is the desired depth
  \ifnum\value{workdepth}>#1% Check if we are deeper than desired
    \addtocounter{workdepth}{-1}
    \end{itemize}
    \checkdepth{#1} % Keep going until desired depth
  \fi
  \ifnum\value{workdepth}<#1% Check if we are shallower than desired
    \ifnum\value{workdepth}<2
      \begin{itemize}[nosep,itemindent=-1cm,label=]
    \fi
    \ifnum\value{workdepth}=2
      \begin{itemize}[nosep,itemindent=-1cm,label=$\bullet$]
    \fi
    \stepcounter{workdepth}
    \checkdepth{#1}% Keep going until desired depth
  \fi
  % Nothing needs to be done if we are at desired depth
}
